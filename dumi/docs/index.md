## 安装
```jsx| pure
$ npm install ithink --save

$ yarn add ithink
```

## 使用
```jsx| pure
import { Button } from "ithink";

const App = () => (
     <Button onClick={handlerSyncClick} loading={true}  >
        异步按钮
      </Button>
);
```
